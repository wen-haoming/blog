# es6的Generator生成器函数

es6提供另一种的异步编程方式,生成器函数,他的返回值是一个迭代对象,并且这个迭代器对象是针对这个生成器函数的内部的状态进行遍历

generator迭代器函数有2个特征

  1. 需要在function关键字和函数名之间有一个*符号
  2. 函数内部能够识别yield表达式,定义不同的状态
  
  function * test(){
      console.log("test")
      yield "a"
      yield "b"
      return 1
  }
  //该函数定义了3个状态, 分别是a，b，1
  // 当函数执行后内部的逻辑并不会立即执行
  
  // 需要执行该函数返回的迭代器对象next方法
  let it = test()
   console.log(it.next())  //  test   { value: 'a', done:false }
   console.log(it.next()) // { value: 'b', done: false}
   console.log(it.next()) // { value: 1, done: true }
 
  function* test () {
  let aa = yield 'a'
  console.log(aa, '----')
  let bb = yield 'b'
  console.log(bb, '----')
  let cc = yield 'c'
  console.log(cc, '----')
  return 'd'
}
  let it = test(123)
  console.log(it.next(1))
  console.log(it.next(2))
  console.log(it.next(3))
  console.log(it.next(4))
  console.log(it.next(5))
    
   // { value: 'a', done: false }
   //  2 ----
   // { value: 'b', done: false }
   // 3 ----
   // { value: 'c', done: false }
   // 4 ----
   // { value: 'd', done: true }
   // { value: undefined, done: true }

综上所述 1. generator生成器函数是分段执行的 2. 代码的每次执行遇到yield或return都会停止下一步的操作 3. 每次的返回迭代器对象的value值都是yield所接收的值 4. 如果done是true就是流程执行完 5. next函数的参数为下一个yield的返回值,注意执行顺序的问题。

# Symbol.iterator属性

  1. [] ,arguments, nodeList, set, map, 等数组结构都有一个Symbol.iterator的属性, 唯独for of 的循环操作的时没有
  2. 如果我们对以上的数据结构进行遍历,会优先使用Symbol.iterator的属性
  3. 并且该属性每次调用都是返回一个迭代器对象
  let obj = {
    1: 'a', 2: 'b', [Symbol.iterator]: function () {
      return {
        next () { // 会不断调用这个next方法知道done为true为止。
          return {
            value: 'xiaoming', 
            done: false
          }
        },
      }
    }
  }

# 如何实现一个能Symbol.iterator属性调用的方法来供对象使用？

  1. 使用原声的方法来模拟迭代器效果
    let obj = {

# 利用es6的生成器函数来模拟迭Symbol.iterator的效果

每次都返回迭代器也就是说 Symbol.iterator属性就是一个生成器函数,每次调用都会返回一个迭代器对象

  let obj = {
  0: 'a', 1: 'b', 2: 'c', length: 3, [Symbol.iterator]: function* () {
    let index = 0;
    let len = this.length;
    while (index !== len) {
      yield this[index++]
    }
  }
}

# 使用迭代器来解决异步问题的方法

每一次的异步迭代的结构都"丢"到下一次的yield里面，但是丢的过程非常麻烦。

  let fs = require("fs").promises
  console.log(fs.readFile("./a.txt"))

  function* test () {
    let a = yield fs.readFile("./a.txt")
    let b = yield fs.readFile(a)
    return b
  }
  let it = test()
  // it.next()
  // console.log()
  it.next().value.then(data => {
    it.next(data.toString()).value.then((data) => {
      console.log(data.toString())
    })
  })

# 使用co库来解决异步问题的方法

  npm install co --save
  let co = require("co")
  co(test).then(data=>{
      console.log(data) // 输出结果
  })

# co库的核心原理

  function co (generator) {
    return new Promise((resolve, reject) => {
      let it = generator()
      let next = function (data) {
        let { value, done } = it.next(data)
        if (!done) {
          Promise.resolve(value).then(data => { // 万一如果value是promise需要对promise进行解析取最后的返回值
            next(data)
          })
        } else {
          resolve(value)
        }
      }
      next()
    })
}

可以看出实现起来并不困难,可以总结几个关键点

  1. 返回一个promise对象,实际上就是对Promise的api的再次封装
  2. 封装next函数来迭代每一次的结果
  3. yield接受的值如果是promise对象的话,那么必须对此promise进行解析操作
